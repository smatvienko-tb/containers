explain analyze select count(*) from t1;

set jit_above_cost = 10; set jit_inline_above_cost = 10; set jit_optimize_above_cost = 10;
explain analyze select count(*) from t1;

explain analyze select avg(sub.aid), sum(sub.bid), COUNT(*) from ( select a.id aid, b.id bid from t1 a INNER join t1 b on a.id < b.id AND (b.id - a.id <= 3 ) where a.id > -150 AND b.id % 2 = 1) sub;

RESET jit; RESET jit_above_cost; RESET jit_inline_above_cost; RESET jit_optimize_above_cost;
